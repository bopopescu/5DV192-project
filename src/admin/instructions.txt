# 1. login to vm
ssh {IP}

# 2. generate keys
ssh-keygen

# 3. copy the public key to github.com
cat ~/.ssh/id_rsa.pub

# 4. clone project
cd ~
git clone git@github.com:nygrenkaj/5DV192-project.git

# 5. setup credentials
mkdir ~/.gcloud
touch ~/.gcloud/credentials.json
nano ~/.gcloud/credentials.json

# 6. done
